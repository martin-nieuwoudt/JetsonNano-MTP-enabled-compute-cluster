cd ~/llama.cpp

# Patch 1: Stub cuda_bf16.h (CUDA 10.2 doesn't have it)
cat > ggml/src/ggml-cuda/vendors/cuda_bf16.h << 'BF16EOF'
#pragma once
struct __nv_bfloat16 { unsigned short __x; __host__ __device__ __nv_bfloat16() : __x(0) {} __host__ __device__ __nv_bfloat16(float f) : __x(0) { (void)f; } __host__ __device__ operator float() const { return 0.0f; } };
typedef __nv_bfloat16 nv_bfloat16;
BF16EOF

# Patch 2: Use quotes for local stub header (NVCC finds it in vendors/)
sed -i 's|#include <cuda_bf16.h>|#include "cuda_bf16.h"|' ggml/src/ggml-cuda/vendors/cuda.h

# Patch 3: NVCC 10.2 doesn't allow constexpr __device__
sed -i 's/constexpr __device__/__device__ const/g' ggml/src/ggml-cuda/common.cuh

# Patch 4: NVCC 10.2 doesn't have __builtin_assume
sed -i '1s/^/#ifndef __builtin_assume\n#define __builtin_assume(x) ((void)0)\n#endif\n/' ggml/src/ggml-cuda/fattn-common.cuh

# Patch 5: Fix unique_ptr copy errors (GCC 10 strictness)
sed -i 's/queue\.push(msg)/queue.push(std::move(msg))/g' ggml/src/ggml-rpc/ggml-rpc.cpp
sed -i 's/queue\.push(ev->msg)/queue.push(std::move(ev->msg))/g' ggml/src/ggml-rpc/ggml-rpc.cpp
